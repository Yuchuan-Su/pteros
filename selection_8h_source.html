788  void split(std::vector<Selection>& parts, F callback){
789  using namespace std;
790  parts.clear();
791  using Ret = decltype(callback(*this,0));
792  // Map
793  map<Ret,vector<int> > m;
794  for(int i=0; i<size(); ++i){
795  m[callback(*this,i)].push_back(index(i));
796  }
797  // Create selections
798  typename map<Ret,vector<int> >::iterator it;
799  for(it=m.begin();it!=m.end();it++){
800  parts.emplace_back(*system, it->second.begin(), it->second.end());
801  }
802  }
